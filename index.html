Juego de 2 Dados

Este programa esta fue dise침ado para elegir simular 2 dados los cuales pueden arrojar un resultado entre 1 y 6 y mostrarlos en la primera fila de una pantalla LCD. Ademas tambien sumara el resultado de los tres dados mostrando haci en la segunda fila el resultado final.

TINKERCAD

El conexionado simulado en el Tinkercad en el cual utilizamos una Placa Arduino Uno R3, un Potenciometro, un LCD 16x2, un Pulsador, una Resistencia y Cables para hacer las conexiones de diferentes colores para que sea mas facil identificar su funcionalidad en el circuito.

#include< LiquidCrystal.h >
const int RS=3,E=5,D4=8,D5=9,D6=10,D7=11;
LiquidCrystal lcd(RS,E,D4,D5,D6,D7);
const int pulsador = 12;
bool presionado = LOW;
bool NO_presionado = HIGH;
void setup()
{
lcd.begin(16,2);
pinMode(pulsador,INPUT);
lcd.clear();
lcd.setCursor(1,0);
lcd.print("JUEGO DE DADOS");
delay(2000);
}
void loop(){
if(digitalRead(pulsador) == presionado){
delay(10);
volver:
if(digitalRead(pulsador) != presionado){
lcd.clear();
int acum=0;
lcd.setCursor(1,0);
lcd.print("VALORES");
lcd.print(" ");
int k=9;
for(int c
ontador=0;contador<3;contador++){
int valor=random(1,7);
lcd.setCursor(k,0);
lcd.print(valor);
lcd.print(" ");
acum=acum+valor;
k=k+2;
}
lcd.setCursor(1,1);

lcd.print(" ");
lcd.setCursor(12,1);
